// alan-ide/frontend/src/components/ConceptFieldCanvas.tsx
import React, { useEffect, useRef } from "react";
import * as d3 from "d3";

type ConceptNode = {
  id: string;
  label: string;
  x: number;
  y: number;
  phase: number; // [0, 2œÄ] ‚Üí mapped to hue
};

type ConceptEdge = {
  source: string;
  target: string;
};

type ConceptGraph = {
  nodes: ConceptNode[];
  edges: ConceptEdge[];
};

export const ConceptFieldCanvas: React.FC = () => {
  const ref = useRef(null);

  useEffect(() => {
    fetch("http://localhost:8000/concept_graph")
      .then((res) => res.json())
      .then((graph: ConceptGraph) => {
        const svg = d3.select(ref.current);
        svg.selectAll("*").remove();

        const width = 800;
        const height = 600;
        svg.attr("width", width).attr("height", height);

        const line = svg
          .selectAll("line")
          .data(graph.edges)
          .enter()
          .append("line")
          .attr("x1", (d) => graph.nodes.find((n) => n.id === d.source)?.x || 0)
          .attr("y1", (d) => graph.nodes.find((n) => n.id === d.source)?.y || 0)
          .attr("x2", (d) => graph.nodes.find((n) => n.id === d.target)?.x || 0)
          .attr("y2", (d) => graph.nodes.find((n) => n.id === d.target)?.y || 0)
          .attr("stroke", "#ccc");

        svg
          .selectAll("circle")
          .data(graph.nodes)
          .enter()
          .append("circle")
          .attr("cx", (d) => d.x)
          .attr("cy", (d) => d.y)
          .attr("r", 12)
          .attr("fill", (d) => `hsl(${(d.phase / (2 * Math.PI)) * 360}, 100%, 60%)`);

        svg
          .selectAll("text")
          .data(graph.nodes)
          .enter()
          .append("text")
          .attr("x", (d) => d.x + 15)
          .attr("y", (d) => d.y + 5)
          .text((d) => d.label)
          .style("font-size", "12px");
      });
  }, []);

  return <svg ref={ref}></svg>;
};

class MorphRequest(BaseModel):
    node_id: str
    epsilon: float
